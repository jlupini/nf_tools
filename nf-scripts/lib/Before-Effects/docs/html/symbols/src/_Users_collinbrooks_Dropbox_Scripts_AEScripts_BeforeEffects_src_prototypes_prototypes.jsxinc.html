 69      * Determines whether or not a given value is within a given array.
 70      * @name inArray
 71      * @memberOf Array
 72      * @param {Any} needle The needle to search for
 73      * @param {Array} haystack The array (haystack) to search for the needle in.
 74      * @returns {Array} True if the needle is found, false if it isn't
 75      * TODO: Verify the haystack is truly an Array Object!
 76      */
 77     function (needle, haystack) {
 78         var i;
 79         for (i in haystack)
 80         {
 81             if (haystack[i] === needle)
 82             {
 83                 return true;
 84             }
 85         }
 86         return false;
 87     }

 90 if (!Array.hasOwnProperty("filter")) {
 91     /**
 92      * Filters an array based upon a filter function.
 93      * @memberOf Array
 94      * @param {Array} a The array to filter
 95      * @param {Function} f The function to use when filtering the array. If the
 96      * function returns true, the item in the array is included in the final
 97      * filtered list. If it is false, it is ignored. This function should allow
 98      * for one argument only.
 99      * @returns {Array}
100      *TODO:
101      * Needs to conform to ECMA-262 5th edition (15.4.4.20)
102      */
103     Array.prototype.filter = function (a, f) {
104         var newArray = [], i;
105         for (i in a)
106         {
107             if (f(i))
108             {
109                 newArray[newArray.length] = i;
110             }
111         }
112         return newArray;
113     };
