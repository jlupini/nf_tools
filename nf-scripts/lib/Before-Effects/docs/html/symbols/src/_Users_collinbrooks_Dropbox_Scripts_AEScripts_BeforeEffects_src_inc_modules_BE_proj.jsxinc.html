 83 /**
 84  *Returns whether or not a composition with the given name exists.
 85  *@param {String} n The name of the composition to search for
 86  *@param {Bool} [c=false] Whether or not to use the compositionCache value. Useful
 87  *when you have recently traversed the project to find all of the
 88  *composition items and you know none have been added since then.
 89  *@returns {Bool|null} True if the composition exists, false if it doesn't.
 90  *null if there is not a project defined.
 91  */
 92 BE.proj.compExists = function (n, c) {
 93 	c = c || false;
 94 	var p = BE.proj.getProject(),
 95 		l = BE.proj.getCompList(c);
 96 
 97 	if (p !== null)
 98 	{
 99 		return l.inArray(n, l);
100 	} else {
101 		return null;
102 	}
103 };

197 /**
198  *Convenience function that makes sure a project and/or an activeItem are
199  *defined. If both are undefined, false is returned. If one or the other
200  *is undefined, false is returned as the value within the returned array.
201  *@return {Bool|Array} A zero based array containing a reference to the
202  *project object and activeItem respectively or False if proj and activeItem
203  *are undefined.
204  *@example
205  *if (v = BE.proj.verifyProjAndActiveItem()) {
206  *  proj = v[0];
207  *  comp = v[1];
208  *}
209  */
210 BE.proj.verifyProjAndActiveItem = function () {
211 	var proj = BE.proj.getProject(),
212 		aItem = BE.proj.getActiveComp();
213 	BE.log.debug("Active Comp " + aItem.name);
214 	if (!proj && !aItem)
215 	{
216 		BE.log.debug(BE.ERROR.NO_OPEN_PROJECT);
217 		return false;
218 	} else {
219 		return [proj, aItem];
220 	}
221 };

223 /**
224  * Gathers an array of compositions that are within the project.
225  * @param {Bool} c Whether or not to use the cached array of the project's
226  * compositions. This is useful when you are calling this method back to
227  * back and you don't need to grab the current list of compositions.
228  * @return {CompItem[]|Bool} If there were no errors, an array of
229  * compositions is returned. Otherwise, false is returned.
230  */
231 BE.proj.getCompList = function (c) {
232 	BE.log.debug("Getting Comp List");
233 	var p, i, v, comps = [], list, refillCache = false,
234 		//Because app.project.items is stupid and its index is 1 based,
235 		//and because we are possibly going to be using the cache array
236 		//(which is 0 based), we need to create a variable that keeps track
237 		//of what our index offset should be. Default is 1 for non-cache
238 		//lists.
239 		indexOffset = 1;
240 
241 	v = BE.proj.verifyProjAndActiveItem();
242 
243 	if (v && v[0])
244 	{
245 		p = v[0];
246 
247 		//Determine whether or not to use the cached values or not.
248 		if (c && BE.proj.compositionCache.length !== 0)
249 		{
250 			BE.log.debug("Using compositionCache!");
251 			list = BE.proj.compositionCache;
252 			indexOffset = 0; //The cache is zero based
253 		} else {
254 			list = p.items;
255 
256 			//Clear the composition cache since we are going to traverse the
257 			//items within the project
258 			BE.proj.compositionCache = [];
259 			refillCache = true;
260 		}
261 
262 		BE.log.debug("Current Project: " + p.name);
263 
264 		BE.log.debug("Going through the list of items: Length = " +
265 				list.length);
266 		for (i = 0; i < list.length; i += 1)
267 		{
268 			BE.log.write(BE.log.types.DEBUG,
269 					"Item " + (i + indexOffset) + ":" +
270 					list[i + indexOffset].typeName + " " +
271 					list[i + indexOffset].name);
272 			if (list[i + indexOffset] instanceof CompItem)
273 			{
274 				comps.push(list[i + indexOffset]);
275 
276 				if (refillCache)
277 				{
278 					BE.proj.compositionCache.push(
279 							list[i + indexOffset]);
280 				}
281 			}
282 		}
283 		BE.log.debug("Returning " +
284 				comps.length + " compositions.");
285 		return comps;
286 	} else {
287 		return false;
288 	}
289 };

291 /**
292  * Grab the first composition found in the project's items
293  * with the given name at an optionalnly given root folder.
294  * @param {String} n The name of the composition to find.
295  * @param {Folder} [f] The folder to use as the root of the search. This is
296  * helpful if you know you don't need to search through the entire project but
297  * only a specific folder for one.
298  * @returns {CompItem|null} Returns the first composition that is found
299  * within the project with the given name or null if none is found.
300  */
301 BE.proj.getCompByName = function (n, f) {
302 	var f = (f) ? f : BE.proj.getProject(),
303 		i;
304 
305 	if (f !== null)
306 	{
307 		for (i = 1; i <= f.items.length; i += 1)
308 		{
309 			if (f.items[i] instanceof CompItem &&
310 					f.items[i].name === n)
311 			{
312 				return f.items[i];
313 			}
314 		}
315 
316 		return null;
317 
318 	} else {
319 		throw (new Error("UndefinedProject Error: Could not get " +
320 					"composition by name because no project is " +
321 					"defined!"));
322 	}
323 };

465 /**
466  *Get an optionally filtered list of the items within the active project.
467  *@param {String|Array} [items] The typeName (or an array of typeNames)of
468  *the Item to include in the returned values.
469  *
470  *Example: "Composition", "Folder", "Footage"
471  *@param {Bool} c Whether or not to utilize the cached items that were found
472  *when this method was previously executed.
473  *@returns {Array} A 0 based array of the filtered items.
474  */
475 BE.proj.getItemList = function (items, c) {
476 	var p = BE.proj.getProject(),
477 		i,
478 		collection = [];
479 
480 	//Because app.project.items is stupid and its index is 1 based,
481 	//and because we are possibly going to be using the cache array
482 	//(which is 0 based), we need to create a variable that keeps track
483 	//of what our index offset should be. Default is 1 for non-cache
484 	//lists.
485 	//indexOffset = 1;
486 
487 	//Is a project open?
488 	if (p !== false)
489 	{
490 		//Does the user want to filter the list?
491 		if (items)
492 		{
493 			//If i is a String, make it an array.
494 			if (items instanceof String)
495 			{
496 				items = [items];
497 			}
498 
499 			//Get the list of items in the project
500 			for (i = 1; i <= p.items.length; i += 1)
501 			{
502 				//See if the current item is in the array that was sent.
503 				if (items.inArray(p.items[i].typeName, items))
504 				{
505 					collection.push(p.items[i]);
506 				}
507 			}
508 		} else {
509 			collection = p.items;
510 		}
511 
512 		return collection;
513 
514 	} else {
515 		return false;
516 	}
517 };
