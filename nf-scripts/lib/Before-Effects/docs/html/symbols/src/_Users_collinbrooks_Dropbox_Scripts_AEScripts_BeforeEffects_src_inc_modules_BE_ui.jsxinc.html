 41 /**
 42  * Takes an error that is returned when a ScriptUI resource specification
 43  * is malformed and makes it easier for you to figure out where the
 44  * error occures.
 45  * @param {Error} err The error that was returned when the resource string
 46  * was run.
 47  * @param {String} res The resource string that contains the error.
 48  * @return {String} The error text that is generated.
 49  */
 50 BE.ui.resError = function (err, res)
 51 {
 52 	/**
 53 	 *@type Regexp
 54 	 */
 55 	var rx,
 56 		search,
 57 		errorLine,
 58 		errorCharOffset,
 59 		lines,
 60 		theLine,
 61 		errText,
 62 		surroundingLinesPadding,
 63 		linesBefore,
 64 		linesAfter,
 65 		l;
 66 
 67 	rx = /Error in line (\d*)\, at character offset (\d*)\,/;
 68 
 69 	//If this error is actually a resource error, handle it like one.
 70 	if (rx.test(err.description)) {
 71 		BE.log.debug("Resource Error occured...");
 72         BE.log.debug("Original error: " + err.description);
 73 		search = rx.exec(err.description);
 74 		errorLine = parseInt(search[1], 10);
 75 		errorCharOffset = parseInt(search[2], 10);
 76 		lines = res.split("\n");
 77 		theLine = lines[errorLine];
 78 
 79 		//Error printing options
 80 		surroundingLinesPadding = 10;
 81 
 82 		//Set the default lines before and after based upon
 83 		//the surroundingLinesPadding value
 84 		linesBefore = surroundingLinesPadding;
 85 		linesAfter = surroundingLinesPadding;
 86 
 87 
 88 		BE.log.debug("Error Line # = " + errorLine);
 89 		BE.log.debug("Character Offset = " + errorCharOffset);
 90 		BE.log.debug("Lines = " + lines.length);
 91 		BE.log.debug("The Line Text = " + theLine);
 92 
 93 		//If the error line is in lines 0 to surroundingLinesPadding,
 94 		//set linesbefore to maximum allowed
 95 		if (errorLine <= surroundingLinesPadding) {
 96 			linesBefore = errorLine;
 97 
 98 			BE.log.debug("Lines Before = " + linesBefore);
 99 		}
100 
101 		if (errorLine + surroundingLinesPadding > lines.length) {
102 			linesAfter = lines.length - errorLine;
103 			BE.log.debug("Lines After Changed To: " + linesAfter);
104 		}
105 
106 		errText = "Original Error: \n " + err.description + "\n" +
107             "Line: " + errorLine + "; " + "Character Offset: " +
108 			errorCharOffset + "\n\n";
109 
110 		//Print the before lines
111 		for (l = linesBefore; l < errorLine; l += 1) {
112 			errText += l + lines[l] + "\n";
113 		}
114 
115 		errText += errorLine + theLine + "\n";
116 		BE.log.debug("Lines After = " + linesAfter);
117 
118 		//Print the after lines
119 		for (l = errorLine + 1; l < errorLine + linesAfter; l += 1)
120 		{
121 			errText += l + lines[l] + "\n";
122 		}
123 
124 		BE.log.debug("Error = " + errText);
125 
126 		BE.alertError(errText);
127 	} else {
128 		BE.log.debug("Misc Resource Error occured...");
129 		//This error is not an error caused by malformed resource specs.
130 		//Alert the error as a normal error and include the resource string.
131 		BE.alertError(err.description + "\n Resource String: \n\n" + res);
132 	}
133 	throw(new Error("Fatal UI Resource String Error. Halting execution."));
134 };

136 /**
137  * Aids in the creation of a label/input pair.
138  *
139  * @param {Object} d An object filled with properties that describe
140  * different aspects or values within this label/input pair.
141  * @param {String} d.groupName The name to give the group.
142  * @param {String} d.labelName The name to give the label of the input..
143  * @param {String} d.inputName The name to give the input field.
144  * @param {String} [d.groupOrientation="row"] The orientation of the group.
145  * Possible values are 'row' or 'col'.
146  * @param {String} [d.groupMargins=0] The margins to set for the group.
147  * @param {String} [d.groupAlignment="['fill','fill']"] The alignment to
148  * use for the group.
149  * @param {String} [d.inputText=""] The text to use as the input's value.
150  * @param {String} [d.inputMinSize="[]"] The minimum size for the
151  * input field.
152  * @param {String} [d.inputJustify="right"] How the label text is
153  * to be justified.
154  * @param {String} [t="EditText"] The type of input to use.
155  * Ex. DropDown, EditText.
156  * @param {String} [extraRes=""] A resource string that is inserted
157  * after the input field.
158  * @return {String|False} Returns a resource string or false if
159  * there was an error.
160  */
161 BE.ui.generateInputField = function (d, t, extraRes)
162 {
163 
164 	/*
165 	 * Returns a defined value
166 	 * Checks to see if 'a' is undefined. If so, b is returned. Otherwise,
167 	 * 'a' is returned.
168 	 * @param {Any} a Contains the value that should be checked as undefined
169 	 * or not.
170 	 * @param {Any} b The value to return if 'a' is undefined.
171 	 * @return {Any} Returns 'a' if 'a' is not undefined, otherwise, it
172 	 * returns 'b'.
173 	 */
174 	var getDefined = function (a, b) {
175 		return (a === undefined) ? b : a;
176 	},
177 
178 		e = false,
179 		tempD,
180 		r;
181 
182 	//Handle some errors
183 	if (d === undefined)
184 	{
185 		BE.alertError(
186 				new Error(BE.ui.error.NO_DATA_FOR_INPUT_GENERATOR,
187 					$.fileName)
188 				);
189 		e = true;
190 	} else if (d.groupName === undefined) {
191 		BE.alertError(new Error(BE.ui.error.NO_GROUP_NAME,
192 					$.fileName));
193 		e = true;
194 	} else if (d.labelName === undefined) {
195 		BE.alertError(new Error(BE.ui.error.NO_LABEL_NAME,
196 					$.fileName));
197 		e = true;
198 	} else if (d.inputName === undefined) {
199 		BE.alertError(new Error(BE.ui.error.NO_INPUT_NAME,
200 					$.fileName));
201 		e = true;
202 	}
203 
204 	if (!e)
205 	{
206 		//If the type of input was not specified, use EditText
207 		if (t === undefined)
208 		{
209 			t = "EditText";
210 		}
211 		//Create a template object that should be used if a value
212 		//is undefined
213 		tempD = {
214 groupOrientation: "row",
215 				  groupMargins: 0,
216 				  groupAlignment: "['fill','fill']",
217 				  inputText: "",
218 				  inputMinSize: "[]",
219 				  inputAlignment: "['fill', 'middle']",
220 				  inputJustify: "right"
221 		};
222 
223 		d.groupOrientation	= getDefined(d.groupOrientation,
224 				tempD.groupOrientation);
225 		d.groupMargins		= getDefined(d.groupMargins,
226 				tempD.groupMargins);
227 		d.groupAlignment	= getDefined(d.groupAlignment,
228 				tempD.groupAlignment);
229 		d.inputText			= getDefined(d.inputText,
230 				tempD.inputText);
231 		d.inputMinSize		= getDefined(d.inputMinSize,
232 				tempD.inputMinSize);
233 		d.inputAlignment	= getDefined(d.inputAlignment,
234 				tempD.inputAlignment);
235 		d.inputJustify		= getDefined(d.inputJustify,
236 				tempD.inputJustify);
237 
238 		r = "";
239 
240 		r += d.groupName + ":" +"Group {" +
241 			"orientation: '" + d.groupOrientation + "', " +
242 			"margins: " + d.groupMargins + ", " +
243 			"alignment: " + d.groupAlignment + ", " +
244 			d.labelName + ": StaticText {" +
245 			"text: '" + d.labelText + "'" +
246 			"}," + //End d.labelName
247 			d.inputName + ": " + t + " {";
248 
249 		r += (t === "EditText") ?
250 			"text: '" + d.inputText + "'," :
251 			"";
252 		r += "minimumSize: " + d.inputMinSize + ", " +
253 			"alignment: " + d.inputAlignment + ", " +
254 			"justify: '" + d.inputJustify +
255 			"'},"; //End d.inputName
256 
257 		r += (extraRes !== undefined) ? extraRes : "";
258 		r += "},";
259 		return r;
260 	} else {
261 		return e;
262 	}
263 };

265 /**
266  * @class Takes a given window or palette object and assits in the creation of
267  * different views for the UI.
268  *
269  * The power comes in its ability to keep track of and give easy access to
270  * the hidden/shown views. Views are offered back to the user as a named
271  * array of references to the base Group object of each view.
272  * @param {Window|Palette} w The Window or Palette to base this window off of.
273  * @param {String} windowProps A string of different properties that should
274  * be set within the main group of the Window|Palette. Commas need only be
275  * present between defined properties, not at the end.
276  * 
277  * Example:
278  *
279  * "orientation: 'column', alignment: ['left', 'middle']"
280  * @returns {ViewsConstructor} The ViewsConstructor object that contains all
281  * of the information about created Window or Palette. The 'build' method
282  * returns the actual window resource that should be used.
283  *
284  */
285 BE.ui.viewsConstructor = function (w, windowProps) {
286 
287 	var views = [],
288 		viewReferences = {},
289 		built = false,
290 
291 		/**
292 		 *Formats a resource string
293 		 *@param {String} r The resource string to format
294 		 *@returns {String} The formatted resource string.
295 		 */
296 		formatRes = function (r) {
297 			//Replace { with {\n
298 			r = r.replace(/{/g, "{\n");
299 				r = r.replace(/},/g, "},\n");
300 				return r;
301 		},
302 
303 		/**
304 		 * Returns the final resource string of this ViewConstructor.
305 		 * @returns {String} The resource string of the window constructed by
306 		 * this ViewConstructor wrapped within group{} and with the window
307 		 * properties set upon creation of this ViewConstructor.
308 		 */
309 		finalResource = function () {
310 			var i,
311 			viewResources = "";
312 
313 			for (i = 0; i < views.length; i += 1) {
314 				viewResources += views[i].res;
315 
316 				if (i !== views.length - 1) {
317 					viewResources += ",\n";
318 				}
319 			}
320 
321 			return "group {\n" +
322 				windowProps + ",\n" +
323 				viewResources + "\n" +
324 				"}";
325 		};
326 
327 		return {
328 			/**
329 			 * Add a view to this ViewConstructor. This view will have its
330 			 * visibility determined by the 'showView' method and will be
331 			 * displayed as a stack within the main window.
332 			 * @param {String} n The name of the view. This name will be used
333 			 * when accessing this view through the array returned by the
334 			 * 'getViews' method.
335 			 * @param {String} res The resource string that defines the content
336 			 * of this view. This resource string will be encapsulated
337 			 * by Group{} and will have the name given as the 'n' argument.
338 			 *
339 			 * Note: Do not put a trailing comma after the resource string for
340 			 * this view. It is automatically added.
341 			 *
342 			 * @returns Nothing.
343 			 * @extends BE#ui#viewConstructor
344 			 * @see getViews
345 			 * @see showView
346 			 * 
347 			 */
348 			"addView": function (n, res) {
349 				views[views.length] = {
350 					"name": n,
351 					"res": n + ": Group {\n" +
352 						formatRes(res) +
353 						"}"
354 				};
355 			},
356 				/**
357 				 * Creates the resource string made up of all the different views
358 				 * added by the 'addView' method and returns the Window|Palette
359 				 * Object reference.
360 				 * @returns {Window|Palette} The final Window|Palette created by
361 				 * this ViewConstructor. This should be used to populate the
362 				 * variable you wish to use as your Window|Palette base group.
363 				 * @extends BE.ui.viewConstructor
364 				 * @see addView
365 				 */
366 				"build": function () {
367 					var i,
368 					finalRes = finalResource();
369 
370 					//Add the final resource string to the grp field of the window
371 					try {
372 
373 						w.grp = w.add(finalRes);
374 					} catch (e) {
375 						BE.alertError(finalRes);
376 						BE.ui.resError(e, finalRes);
377 						//                    throw(new Error("ViewConstructor ResourceError: Unable to "+
378 						//                        "build ViewConstructor because of error: " +
379 						//                        e.toString()));
380 					}
381 
382 
383 					//Populate view references
384 					for (i = 0; i < views.length; i += 1) {
385 						viewReferences[views[i].name] = w.grp[views[i].name];
386 					}
387 					built = true;
388 					return w;
389 				},
390 
391 				/**
392 				 * Shows the given view and hides all of the others.
393 				 * @param {String} v The name of the view to show.
394 				 * @returns Nothing.
395 				 * @extends BE.ui.viewConstructor
396 				 */
397 				"showView": function (v) {
398 					if (built === true)
399 					{
400 						var i, currView;
401 
402 						for (i = 0; i < views.length; i += 1) {
403 							currView = w.grp[views[i].name];
404 							if (views[i].name === v) {
405 								currView.show();
406 							} else {
407 								currView.hide();
408 							}
409 						}
410 					}
411 				},
412 				/**
413 				 * Returns an Object of references to the base group of the given
414 				 * views. Useful when you want to access each view's Group Object.
415 				 * @returns {Object of Groups} An Object filled with references of
416 				 * the different view's Group Objects.
417 				 *
418 				 * Example:
419 				 * <code>
420 				 * var w = new Window("palette",
421 				 * BE.PRad.scriptName + " " + BE.PRad.scriptVersion,
422 				 * undefined,
423 				 * {
424 				 *  "resizeable": false
425 				 *  });
426 				 * var viewConstructor = BE.UI_Helper.viewConstructor(w,
427 				 *  "orientation: 'column'"
428 				 * );
429 				 * viewConstructor.addView("viewOne",
430 				 * "alignment:['fill','fill']," +
431 				 * "orientation: 'column'," +
432 				 * "nameGroup: Group {" +
433 				 *  "orientation: 'row', " +
434 				 *  "margins:[10,10,10,0], " +
435 				 *  "nameLabel: StaticText {" +
436 				 *  "text: 'Your Name:'" +
437 				 *  "}");
438 				 * var v = viewConstructor.getViews();
439 				 * var viewOneReference = v.One;
440 				 * alert(viewOneReference.nameGroup.nameLabel.text); //"Your Name:"
441 				 * </code>
442 				 * @extends BE.ui.viewConstructor
443 				 */
444 				"getViews": function () {
445 					return viewReferences;
446 				},
447 
448 				/**
449 				 * Returns the final resource string of the Window|Palette created
450 				 * by this ViewConstructor.
451 				 * @returns {String} The final resource string of the Window|Palette
452 				 * created by this ViewConstructor.
453 				 * @extends BE.ui.viewConstructor
454 				 */
455 				"getRes": function () {
456 					return finalResource();
457 				}
458 		};
459 		};

462 			var r = "group { " + res + "},";
463 			//Add the res to the parent
464 			return parent.add(r).grp;
