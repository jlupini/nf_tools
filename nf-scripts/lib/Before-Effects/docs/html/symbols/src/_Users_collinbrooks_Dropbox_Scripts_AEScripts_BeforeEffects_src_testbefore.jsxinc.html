 55     // Test for equality any JavaScript type.
 56     // Discussions and reference: http://philrathe.com/articles/equiv
 57     // Test suites: http://philrathe.com/tests/equiv
 58     // Author: Philippe RathÃ© <prathe@gmail.com>
 59     // Copied from qunit: http://docs.jquery.com/QUnit
 60     equiv = (function () {
 61 
 62         var innerEquiv, // the real equiv function
 63             callers = [], // stack to decide between skip/abort functions
 64             parents = [], // stack to avoiding loops from circular referencing
 65             callbacks = null;
 66 
 67         // Call the o related callback with the given arguments.
 68         function bindCallbacks(o, callbacks, args) {
 69             var prop = TB.objectType(o);
 70             if (prop) {
 71                 if (TB.objectType(callbacks[prop]) === "function") {
 72                     return callbacks[prop].apply(callbacks, args);
 73                 } else {
 74                     return callbacks[prop]; // or undefined
 75                 }
 76             }
 77         }
 78         
 79         callbacks = (function () {
 80 
 81             // for string, boolean, number and null
 82             function useStrictEquality(b, a) {
 83                 if (b instanceof a.constructor || a instanceof b.constructor) {
 84                     // to catch short annotaion VS 'new' annotation
 85                     // of a declaration
 86                     // e.g. var i = 1;
 87                     //      var j = new Number(1);
 88                     return a == b;
 89                 } else {
 90                     return a === b;
 91                 }
 92             }
 93 
 94             return {
 95                 "string": useStrictEquality,
 96                 "boolean": useStrictEquality,
 97                 "number": useStrictEquality,
 98                 "null": useStrictEquality,
 99                 "undefined": useStrictEquality,
100 
101                 "nan": function (b) {
102                     return isNaN(b);
103                 },
104 
105                 "date": function (b, a) {
106                     return TB.objectType(b) === "date" &&
107                         a.valueOf() === b.valueOf();
108                 },
109 
110                 "regexp": function (b, a) {
111                     return TB.objectType(b) === "regexp" &&
112                         a.source === b.source && // the regex itself
113                         a.global === b.global && // and its modifers (gmi) ...
114                         a.ignoreCase === b.ignoreCase &&
115                         a.multiline === b.multiline;
116                 },
117 
118                 // - skip when the property is a method of an instance (OOP)
119                 // - abort otherwise,
120                 //   initial === would have catch identical references anyway
121                 "function": function () {
122                     var caller = callers[callers.length - 1];
123                     return caller !== Object &&
124                         typeof caller !== "undefined";
125                 },
126 
127                 "array": function (b, a) {
128                     var i, j, loop, len;
129 
130                     // b could be an object literal here
131                     if (TB.objectType(b) !== "array") {
132                         return false;
133                     }   
134                     
135                     len = a.length;
136                     if (len !== b.length) { // safe and faster
137                         return false;
138                     }
139                     
140                     //track reference to avoid circular references
141                     parents.push(a);
142                     for (i = 0; i < len; i += 1) {
143                         loop = false;
144                         for (j = 0; j < parents.length; j += 1) {
145                             if (parents[j] === a[i]) {
146                                 loop = true;//dont rewalk array
147                             }
148                         }
149                         if (!loop && !innerEquiv(a[i], b[i])) {
150                             parents.pop();
151                             return false;
152                         }
153                     }
154                     parents.pop();
155                     return true;
156                 },
157 
158                 "object": function (b, a) {
159                     var i, j, loop,
160                         eq = true, // unless we can proove it
161                         aProperties = [], // collection of strings 
162                         bProperties = []; // collection of strings
163 
164                     // comparing constructors is more strict than using
165                     // instanceof
166                     if (a.constructor !== b.constructor) {
167                         return false;
168                     }
169 
170                     // stack constructor before traversing properties
171                     callers.push(a.constructor);
172                     //track reference to avoid circular references
173                     parents.push(a);
174                     
175                     for (i in a) {
176                     // be strict: don't ensure hasOwnProperty and go deep
177                         loop = false;
178                         for (j = 0; j < parents.length; j += 1) {
179                             if (parents[j] === a[i]) {
180                                 //don't go down the same path twice 
181                                 loop = true;
182                             }
183                         }
184                         aProperties.push(i); // collect a's properties
185 
186                         if (!loop && !innerEquiv(a[i], b[i])) {
187                             eq = false;
188                             break;
189                         }
190                     }
191 
192                     callers.pop(); // unstack, we are done
193                     parents.pop();
194 
195                     for (i in b) {
196                         bProperties.push(i); // collect b's properties
197                     }
198 
199                     // Ensures identical properties name
200                     return eq && innerEquiv(aProperties.sort(),
201                             bProperties.sort());
202                 }
203             };
204         }());
205 
206         innerEquiv = function () { // can take multiple arguments
207             var args = Array.prototype.slice.apply(arguments);
208             if (args.length < 2) {
209                 return true; // end transition
210             }
211 
212             return (function (a, b) {
213                 if (a === b) {
214                     return true; // catch the most you can
215                 } else if (
216                         a === null ||
217                         b === null ||
218                         typeof a === "undefined" ||
219                         typeof b === "undefined" ||
220                         TB.objectType(a) !== TB.objectType(b)
221                         ) {
222                     return false; // don't lose time with error prone cases
223                 } else {
224                     return bindCallbacks(a, callbacks, [b, a]);
225                 }
226 
227             // apply transition with (1..n) arguments
228             }(args[0], args[1])) &&
229                     arguments.callee.apply(
230                         this,
231                         args.splice(1, args.length - 1)
232                     );
233         };
234 
235         return innerEquiv;
236 
237     }());

239     /**
240      * Returns the type of the given object as a lowercase string.
241      * Borrowed from QUnit: http://docs.jquery.com/QUnit
242      * @param {Object} obj An object to get the type of.
243      * @returns {String} A lowercase type string.
244      */
245 	this.objectType = function (obj) {
246 		if (typeof obj === "undefined") {
247             return "undefined";
248 
249 		// consider: typeof null === object
250 		}
251 		if (obj === null) {
252             return "null";
253 		}
254 
255 		var type = Object.prototype.toString.call(obj)
256 			.match(/^\[object\s(.*)\]$/)[1] || '';
257 
258 		switch (type) {
259             case 'Number':
260                 if (isNaN(obj)) {
261                     return "nan";
262                 } else {
263                     return "number";
264                 }
265             case 'String':
266             case 'Boolean':
267             case 'Array':
268             case 'Date':
269             case 'RegExp':
270             case 'Function':
271                 return type.toLowerCase();
272 		}
273 		if (typeof obj === "object") {
274             return "object";
275 		}
276 		return undefined;
277 	};

287         /**
288          * Returns a pluralized string.
289          * @memberOf formatter
290          */
291         pluralize: function (val, base, tail) {
292             var s = val + " " + base;
293 
294             s += (val > 1 || val < -1) ? tail : "";
295 
296             return s;
297         },
298         /**
299          * Outputs a result.
300          * @param {String} type A type string that specifies the type of output
301          * to make.
302          * @param {Object} d A data object with key/value pairs required
303          * for the given type.
304          * @memberOf formatter
305          */
306         output: function (type, d) {
307             var outcome;
308             if (type === "moduleHeader") {
309                 $.writeln(d.moduleName + " " +
310                     "(" + this.pluralize(d.numTests, "test", "s") + ")");
311                 $.writeln("=================");
312                 $.writeln("");
313             } else if (type === "moduleFooter") {
314                 $.writeln("=================");
315                 $.writeln(d.moduleName + " results: " +
316                     "PASSED: " + d.passed + ", " +
317                     "FAILED: " + d.failed);
318                 $.writeln("");
319             } else if (type === "testHeader") {
320                 $.writeln(d.testName + " " + "(" +
321                         this.pluralize(
322                             d.numAssertions,
323                             "assertion", "s"
324                         ) + ")"
325                 );
326                 $.writeln("-----------------");
327                 $.writeln("");
328             } else if (type === "testFooter") {
329                 $.writeln("");
330                 $.writeln("-----------------");
331                 $.writeln(d.testName + " results: " +
332                     "PASSED: " + d.passed + ", " +
333                     "FAILED: " + d.failed);
334                 $.writeln("");
335             } else if (type === "testResult") {
336                 outcome = (d.result === true) ? "PASSED - " : "FAILED - ";
337                 $.writeln(" > " + outcome + d.message);
338 
339                 if(d.expected != undefined) {
340                     $.writeln("\t EXPECTED: " + d.expected);
341                 }
342 
343                 if(d.actual != undefined) {
344                     $.writeln("\t ACTUAL:   " + d.actual);
345                 }
346             } else if (type === "summary") {
347                 $.writeln("##### SUMMARY #####");
348                 $.writeln("Total Modules: " + totalModules);
349                 $.writeln("Total Tests: " + totalTests);
350                 $.writeln("Total Assertions: " + totalAssertions);
351                 $.writeln("Total Passed: " + totalPassed);
352                 $.writeln("Total Failed: " + totalFailed);
353             } else {
354                 alert("Unsupported result type: " + d.resultType);
355             }
356         }

575             totalTests: totalTests,
576             totalAssertions: totalAssertions,
577             totalPassed: totalPassed,
578             totalFailed: totalFailed

774     /**
775      * Borrowed from Qunit: http://docs.jquery.com/QUnit
776      * @ignore
777      */
778 	this.raises = function (block, expected, message) {
779 		var actual, ok = false;
780 	
781 		if (typeof expected === 'string') {
782 			message = expected;
783 			expected = null;
784 		}
785 	
786 		try {
787 			block();
788 		} catch (e) {
789 			actual = e;
790 		}
