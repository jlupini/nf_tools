 26 /**
 27  *Returns an array of the layer names within a given composition optionally
 28  *filtering out certain types of layers.
 29  *@param {CompItem} c The composition whose layer names are to be returned
 30  *@param {String[]} [f] An array of strings representing the type of layer
 31  *to be included in the returned array. If not set, nothing is filtered.
 32  *
 33  *Example: 'LightLayer', 'AVLayer', 'CameraLayer', 'ShapeLayer'
 34  *@return {String[]} An array of the names of the layers within the comp.
 35  */
 36 BE.comp.getLayerNames = function (c, f)
 37 {
 38 	var i, names = [], currentLayer, noFilter,
 39 		r = new Regex("/^\[object (.+)\]/");
 40 
 41 	noFilter = (f) ? false : true;
 42 
 43 	for (i = 1; i <= c.numLayers; i += 1)
 44 	{
 45 		currentLayer = c.layer[i];
 46 		if (!noFilter)
 47 		{
 48 			if ( f.inArray(r.exec(currentLayer.toString())[0], f)) {
 49 				names.push(currentLayer.name);
 50 			}
 51 		} else {
 52 			names.push(currentLayer.name);
 53 		}
 54 	}
 55 	return names;
 56 };

 58 /**
 59  *Returns an Array of the layers within a given composition optionally
 60  *filtering out certain types of layers.
 61  *@param {CompItem} c The composition whose layers are to be returned
 62  *@param {String[]} [f=[]] An array of strings representing the type of layer
 63  *to be included in the returned array. If not set, nothing is filtered.
 64  *
 65  *Example: 'LightLayer', 'AVLayer', 'CameraLayer', 'ShapeLayer'
 66  *@return {Layer[]} An array of the layers within the given comp.
 67  */
 68 BE.comp.getFilteredLayers = function (c, f)
 69 {
 70 	BE.log.debug("Getting filtered Layers...");
 71 	BE.log.debug("Comp Name: " + c.name);
 72 	BE.log.debug("Filter: " + f.toString());
 73 	var i, names = [], currentLayer, noFilter,
 74 		r = /^\[object\ (.+)\]/, currentType;
 75 
 76 	noFilter = (f) ? false : true;
 77 
 78 	for (i = 1; i <= c.numLayers; i += 1)
 79 	{
 80 		currentLayer = c.layer(i);
 81 		BE.log.debug("Current Layer: " + currentLayer.name);
 82 		currentType = r.exec(currentLayer.toString())[1];
 83 		BE.log.debug("Current type: " + currentType);
 84 		if (!noFilter && currentType)
 85 		{
 86 			if ( f.inArray(currentType, f)) {
 87 				BE.log.debug("Adding " + currentLayer.name + " to list");
 88 				names.push(currentLayer.name);
 89 			}
 90 		} else {
 91 			names.push(currentLayer.name);
 92 		}
 93 	}
 94 	BE.log.debug("Returning " + names.length + " layers");
 95 	return names;
 96 };

 98 /**
 99  *Goes through each layer within the given comp and builds an array of
100  *references to all the selected properties within the comp.
101  *@param {CompItem} c The composition to search.
102  *@returns {Object[]|Bool} False if no properties are selected, or an array
103  *of object consisting of the following fields:
104  *
105  *parent: The CompItem that the layer with this property is contained in.
106  *layer: The Layer that this property belongs to.
107  *selectedProperties: An array of the properties that are selected in
108  *this layer.
109  */
110 BE.comp.getAllSelectedProperties = function(c)
111 {
112 	BE.log.debug("Getting All Selected Properties...");
113 	BE.log.debug("Using Comp: " + c.name);
114 
115 	var currentLayer, i, props = [],
116 		/**
117 		 *An object that contains information about a layer and it's
118 		 *selected properties.
119 		 *
120 		 *The Object is made up of the following fields:
121 		 *<ul>
122 		 *  <li><b>parent</b>: The layer's parent composition.</li>
123 		 *  <li><b>layer</b>: A references to the actual layer</li>
124 		 *  <li><b>selectedProperties</b>: An array of the selected properties
125 		 *  from the layer.</li>
126 		 *</ul>
127 		 *@type {Object}
128 		 */
129 		theObj;
130 	BE.log.debug("Num Layers: " + c.numLayers);
131 	for (i = 1; i <= c.numLayers; i += 1)
132 	{
133 		currentLayer = c.layer(i);
134 		if (currentLayer.selectedProperties.length !== 0)
135 		{
136 			BE.log.debug("Current Layer [" + i + "]: " + currentLayer.name);
137 			BE.log.debug("Props.length: " + props.length);
138 			theObj = {
139 				"parent": c,
140 				"layer": currentLayer,
141 				"selectedProperties": currentLayer.selectedProperties
142 			};
143 			BE.log.debug("Selected Properties: " + currentLayer.selectedProperties.length);
144 			props.push(theObj);
145 			BE.log.debug("Props.length updated: " + props.length);
146 		}
147 	}
148 
149 	if (props.length !== 0)
150 	{
151 		BE.log.debug("Returning props array");
152 		return props;
153 	} else {
154 		BE.log.debug("Returning False");
155 		return false;
156 	}
157 }
