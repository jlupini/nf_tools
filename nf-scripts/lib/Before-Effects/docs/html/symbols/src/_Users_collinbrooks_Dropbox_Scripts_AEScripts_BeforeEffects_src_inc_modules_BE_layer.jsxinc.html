 22 /**
 23  * Adds a marker with a comment to a specified layer at a specificed time.
 24  * @param {Layer} l The layer to add the marker to.
 25  * @param {String} time The time to add the marker
 26  * @param {String|Int} txt The text to add as the marker's comment value
 27  * @return {Layer} The layer that was sent. Useful for chainging.
 28  */
 29 BE.layer.addMarker = function (l, time, txt)
 30 {
 31 	l.property("Marker").setValueAtTime(time, new MarkerValue(txt));
 32     return l;
 33 };
 34 
 35 /**
 36  * Deletes a marker on a layer at a specified index.
 37  * @param {Layer} l The layer to delete the marker from
 38  * @param {Int} index The index of the marker to delete
 39  * @return {Layer} The layer that was sent. Useful for chainging.
 40  */
 41 BE.layer.deleteMarker = function (l, index)
 42 {
 43 	l.property("Marker").removeKey(index);
 44     return l;
 45 };
 46 
 47 /**
 48  * Get a marker's index based upon its comment value. Goes through each of
 49  * the markers on the passed layer and returns its index.
 50  * @param {Layer} l The layer to delete the marker from
 51  * @param {String} markerText The text to search for in the comment property
 52  * of the selected layer's marker's. Only exact matches return an index.
 53  * @return {Bool|Int} False if 'markerText' does not match any comments
 54  * within the 'l' layer's markers. The index if a match is found.
 55  */
 56 BE.layer.findMarkerIndex = function (l, markerText)
 57 {
 58 	var i;
 59 
 60 	//Loop through the markers
 61 	for (i = l.property("Marker").numKeys; i >= 1; i -= 1)
 62 	{
 63 		//Does the current marker's comment equal the marker text?
 64 		if (l.property("Marker").keyValue(i).comment === markerText)
 65 		{
 66 			//Return the marker index
 67 			return i;
 68 		}
 69 	}
 70 	return false;
 71 };

 73 /**
 74  * Grabs the time of a marker based upon an index. The time is returned
 75  * in the current user time format.
 76  * @param {Layer} l The layer to search for the marker on.
 77  * @param {Int} markerIndex The index of the marker whose time is to be
 78  * returned.
 79  * @returns {String} The time of the marker found at the given index in the
 80  * current user time format.
 81  */
 82 BE.layer.getMarkerTime = function (l, markerIndex)
 83 {
 84 	BE.log.debug("Layer: " + l.name + "; Marker Index: " + markerIndex);
 85 	BE.log.debug("Actual time: " + l.property("Marker").keyTime(markerIndex));
 86 	BE.log.debug("getMarkerTime returns " + timeToCurrentFormat(l.property("Marker").keyTime(markerIndex),
 87 				l.containingComp.frameRate,
 88 				true
 89 				));
 90 
 91 	return timeToCurrentFormat(l.property("Marker").keyTime(markerIndex),
 92 			l.containingComp.frameRate,
 93 			true
 94 			);
 95 };
